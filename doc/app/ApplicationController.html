def authorize

    redirect_to login_url, :alert => t(:session_login) 
  end
end

def session_expiry
  get_session_time_left
  unless @session_time_left > 0
    session[:user_id] = nil
    redirect_to login_url, :alert => t(:session_timeout) 
  end
end
